import numpy as np

# Class i din cho mt nt trong cy quyt nh
class TreeNode:
    def __init__(self, feature=None, threshold=None, left=None, right=None, value=None):
        self.feature = feature      # c trng s dng  chia d liu
        self.threshold = threshold  # Ngng s dng  chia d liu
        self.left = left            # Con tr ti nt con bn tri
        self.right = right          # Con tr ti nt con bn phi
        self.value = value          # Gi tr ca nt nu l nt l

# Hm tnh ch s Gini  o  thun nht ca nt
def gini(y):
    _, counts = np.unique(y, return_counts=True)  # Tm cc lp v s lng phn t trong mi lp
    gini = 1.0 - sum((count / len(y)) ** 2 for count in counts)  # Tnh ch s Gini
    return gini

class G07DecisionTree():
    def __init__(self, max_depth=10):
        self.max_depth = max_depth
        self.tree = TreeNode()

    # Hm fit tree vi datasets
    def fit(self, X, y):
        self.tree = self.build_tree_(X, y, depth=0, max_depth=self.max_depth)
    
    # Hm d on mt tp d liu
    def predict(self, X):
        return np.array([self.predict_tree_(self.tree, x) for x in X])
    
    # Hm chia d liu theo c trng v ngng
    def split_(self, X, y, feature, threshold):
        left_mask = X[:, feature] <= threshold  # Mt n  ly cc phn t nh hn hoc bng ngng
        right_mask = X[:, feature] > threshold  # Mt n  ly cc phn t ln hn ngng
        return X[left_mask], X[right_mask], y[left_mask], y[right_mask]

    # Hm tm c trng v ngng tt nht  chia d liu
    def best_split_(self, X, y):
        best_gini = 1.0
        best_feature = None
        best_threshold = None
        for feature in range(X.shape[1]):  # Duyt qua tng c trng
            thresholds = np.unique(X[:, feature])  # Tm tt c cc ngng duy nht
            for threshold in thresholds:  # Duyt qua tng ngng
                X_left, X_right, y_left, y_right = self.split_(X, y, feature, threshold)  # Chia d liu
                if len(y_left) == 0 or len(y_right) == 0:  # Nu mt trong hai phn trng, b qua
                    continue
                gini_left = gini(y_left)  # Tnh ch s Gini cho phn bn tri
                gini_right = gini(y_right)  # Tnh ch s Gini cho phn bn phi
                gini_split = (len(y_left) * gini_left + len(y_right) * gini_right) / len(y)  # Tnh ch s Gini trung bnh
                if gini_split < best_gini:  # Nu Gini nh hn, cp nht c trng v ngng tt nht
                    best_gini = gini_split
                    best_feature = feature
                    best_threshold = threshold
        return best_feature, best_threshold
    
    # Hm xy dng cy quyt nh  quy
    def build_tree_(self, X, y, depth=0, max_depth=10):
        if len(np.unique(y)) == 1:  # Nu tt c cc phn t cng mt lp, tr v nt l
            return TreeNode(value=y[0])
        if depth >= max_depth:  # Nu  su t gii hn, tr v nt l
            return TreeNode(value=np.bincount(y).argmax())  # Tr v lp ph bin nht
        feature, threshold = self.best_split_(X, y)  # Tm c trng v ngng tt nht
        if feature is None:  # Nu khng tm c c trng tt, tr v nt l
            return TreeNode(value=np.bincount(y).argmax())  # Tr v lp ph bin nht
        X_left, X_right, y_left, y_right = self.split_(X, y, feature, threshold)  # Chia d liu
        left_child = self.build_tree_(X_left, y_left, depth + 1, max_depth)  # Xy dng nt con bn tri
        right_child = self.build_tree_(X_right, y_right, depth + 1, max_depth)  # Xy dng nt con bn phi
        return TreeNode(feature=feature, threshold=threshold, left=left_child, right=right_child)
    
    # Hm d on gi tr da trn cy quyt nh
    def predict_tree_(self, node, X):
        if node.value is not None:  # Nu l nt l, tr v gi tr ca nt l
            return node.value
        if X[node.feature] <= node.threshold:  # Nu gi tr nh hn hoc bng ngng, duyt cy con bn tri
            return self.predict_tree_(node.left, X)
        else:  # Nu gi tr ln hn ngng, duyt cy con bn phi
            return self.predict_tree_(node.right, X)

def preprocess_text(text):
    # Remove punctuation
    text = ''.join([char for char in text if char not in string.punctuation])
    # Tokenize words
    words = text.split()
    # Remove stopwords and stemming
    stop_words = stopwords.words('english')
    stemmer = SnowballStemmer('english')
    words = [stemmer.stem(word) for word in words if word.lower() not in stop_words]
    return ' '.join(words)

emails['cleaned_text'] = emails['text'].apply(preprocess_text)

M ha vector

Thc hin vic trch xut c trng s dng Vectorizer TF-IDF chuyn i d liu vn bn c lm sch thnh dng s c th c s dng bi cc thut ton hc my.

Vit tt ca thut ng ting Anh term frequency inverse document frequency,tf-idf l trng s ca mt t trong vn bn thu c qua thng k th hin mc quan trng ca t ny trong mt vn bn, m bn thn vn bn ang xt nm trong mt tp hp cc vn bn.

Thut ton ny thng c s dng v: trong ngn ng lun c nhng t xy ra thng xuyn vi cc t khc.

  1. Tf- term frequency : dng c lng tn xut xut hin ca t trong vn bn. Tuy nhin vi mi vn bn th c di khc nhau, v th s ln xut hin ca t c th nhiu hn . V vy s ln xut hin ca t s c chia di ca vn bn (tng s t trong vn bn ).

TF(t, d) = ( s ln t t xut hin trong vn bn d) / (tng s t trong vn bn d)

  1. IDF- Inverse Document Frequency: dng c lng mc quan trng ca t nh th no . Khi tnh tn s xut hin tf th cc t u c coi l quan trng nh nhau. Tuy nhin c mt s t thng c c s dng nhiu nhng khng quan trng th hin ngha ca on vn , v d : T ni: v, nhng, tuy nhin, v th, v vy,
  • Gii t: , trong, trn,
  • T ch nh: y, , nh,
  • V vy ta cn gim i mc quan trng ca nhng t bng cch s dng IDF :

IDF(t, D) = log_e( Tng s vn bn trong tp mu D/ S vn bn c cha t t )

TfidfVectorizer l mt k thut trch xut c trng t th vin scikit-learn, chuyn i d liu vn bn thnh cc c trng s da trn im s TF-IDF. TF-IDF o lng tm quan trng ca mt t trong mt ti liu so vi s xut hin ca n trong ton b tp hp ti liu. K thut ny gip gim trng s ca nhng t thng gp (nh "ci," "v," v.v.) m t thng tin hn.

Decision Tree (cy quyt nh) v thut ton ID3

C s l thuyt v cy quyt nh

Decision Tree (cy quyt nh) l mt m hnh hc my a ra kt qu da trn cc cu hi. N c dng nh hnh bn di

Trong decision tree, cc mu xm, lc, trn hnh trn c gi l cc node. Cc node th hin u ra (mu lc v ) c gi l node l (leaf node hoc terminal node). Cc node th hin cu hi l cc non-leaf node.

Non-leaf node trn cng (cu hi u tin) c gi l node gc (root node). Cc non-leaf node thng c hai hoc nhiu node con (child node). Cc child node ny c th l mt leaf node hoc mt non-leaf node khc. Cc child node c cng b m c gi l sibling node. Nu tt c cc non-leaf node ch c hai child node, ta ni rng l mt binary decision tree (cy quyt nh nh phn).

Cc cu hi trong binary decision tree u c th a c v dng cu hi ng hay sai. Cc decision tree m mt leaf node c nhiu child node cng c th c a v dng mt binary decision tree. iu ny c th t c v hu ht cc cu hi u c th c a v dng cu hi ng sai.

Thut ton ID3

tng chnh

Trong ID3, chng ta cn xc nh th t ca thuc tnh cn c xem xt ti mi bc. Vi cc bi ton c nhiu thuc tnh v mi thuc tnh c nhiu gi tr khc nhau, vic tm c nghim ti u thng l khng kh thi. Thay vo , mt phng php n gin thng c s dng l ti mi bc, mt thuc tnh tt nht s c chn ra da trn mt tiu chun no (chng ta s bn sm). Vi mi thuc tnh c chn, ta chia d liu vo cc child node tng ng vi cc gi tr ca thuc tnh ri tip tc p dng phng php ny cho mi child node. Vic chn ra thuc tnh tt nht mi bc nh th ny c gi l cch chn greedy (tham lam). Cch chn ny c th khng phi l ti u, nhng trc gic cho chng ta thy rng cch lm ny s gn vi cch lm ti u. Ngoi ra, cch lm ny khin cho bi ton cn gii quyt tr nn n gin hn.

Sau mi cu hi, d liu c phn chia vo tng child node tng ng vi cc cu tr li cho cu hi . Cu hi y chnh l mt thuc tnh, cu tr li chnh l gi tr ca thuc tnh . nh gi cht lng ca mt cch phn chia, chng ta cn i tm mt php o.

Trc ht, th no l mt php phn chia tt? Bng trc gic, mt php phn chia l tt nht nu d liu trong mi child node hon ton thuc vo mt classkhi child node ny c th c coi l mt leaf node, tc ta khng cn phn chia thm na. Nu d liu trong cc child node vn ln vo nhau theo t l ln, ta coi rng php phn chia cha thc s tt. T nhn xt ny, ta cn c mt hm s o tinh khit (purity), hoc vn c (impurity) ca mt php phn chia. Hm s ny s cho gi tr thp nht nu d liu trong mi child node nm trong cng mt class (tinh khit nht), v cho gi tr cao nu mi child node c cha d liu thuc nhiu class khc nhau.

Mt hm s c cc c im ny v c dng nhiu trong l thuyt thng tin l hm entropy. Hm s entropy c din t trong hnh di

Thut ton

Trong ID3, tng c trng s ca entropy ti cc leaf-node sau khi xy dng decision tree c coi l hm mt mt ca decision tree . Cc trng s y t l vi s im d liu c phn vo mi node. Cng vic ca ID3 l tm cc cch phn chia hp l (th t chn thuc tnh hp l) sao cho hm mt mt cui cng t gi tr cng nh cng tt. Nh cp, vic ny t c bng cch chn ra thuc tnh sao cho nu dng thuc tnh phn chia, entropy ti mi bc gim i mt lng ln nht. Bi ton xy dng mt decision tree bng ID3 c th chia thnh cc bi ton nh, trong mi bi ton, ta ch cn chn ra thuc tnh gip cho vic phn chia t kt qu tt nht. Mi bi ton nh ny tng ng vi vic phn chia d liu trong mt non-leaf node. Chng ta s xy dng phng php tnh ton da trn mi node ny.

Trc ht, ti mi node:

Xt mt bi ton vi $C$ class khc nhau. Gi s ta ang lm vic vi mt non-leaf node vi cc im d liu to thnh mt tp $\mathcal{S}$ vi s phn t l $N$. Trong cc im ny c $N_c$ im thuc class $c$, vi mi $c\in \overline{1,C}$. Xc sut mi im ri vo class $c$ c xp x bi $\dfrac{N_c}{N}$. Nh vy entropy ti node ny l \begin{align*} H(\mathcal{S}) = -\sum_{c=1}^C \frac{N_C}{N}\log{\frac{N_c}{N}}\tag{1} \end{align*} Tip theo, gi s thuc tnh c chn chia d liu l $x$. Da trn $x$, cc im d liu trong $\mathcal{F}$ c phn ra thnh $K$ childnode l $\mathcal{S}_1,...,\mathcal{S_K}$ vi s im trong mi childnode l $m_1,...,m_K$. Ta tnh i lng o mc vn c ca mt php chia da vo $x$ trn $\mathcal{S}$ l tng c trng s cc entropy ti $\mathcal{S}_k$ (c tnh nh (1)). \begin{align*} H(x, \mathcal{S}) = \sum_{k=1}^K\frac{m_k}{N}H(S_k) \end{align*} V ta nh ngha gain ca node $\mathcal{S}$ da trn vic phn chia theo thuc tnh $x$ l \begin{align*} G(x, \mathcal{S}) = H(\mathcal{S}) - H(x, \mathcal{S}) \end{align*} Trong thut ton ID3, ti mi node khi phn chia, thuc tnh $x$ c chn phn chia l thuc tnh m c $G(x,\mathcal{S})$ t gi tr ln nht.

im dng ca thut ton:

Trong cc thut ton decision tree ni chung v ID3 ni ring, nu ta tip tc phn chia cc node cha tinh khit, ta s thu c mt tree m mi im trong tp hun luyn u c d on ng (gi s rng khng c hai input ging nhau no cho output khc nhau). Khi , tree c th s rt phc tp (nhiu node) vi nhiu leaf node ch c mt vi im d liu. Nh vy, nhiu kh nng overfitting s xy ra.

trnh overfitting, mt trong s cc phng php sau c th c s dng. Ti mt node, nu mt trong s cc iu kin sau y xy ra, ta khng tip tc phn chia node v coi n l mt leaf node:

  • nu node c entropy bng 0, tc mi im trong node u thuc mt class.

  • nu node c s phn t nh hn mt ngng no . Trong trng hp ny, ta chp nhn c mt s im b phn lp sai trnh overfitting. Class cho leaf node ny c th c xc nh da trn class chim a s trong node.

  • nu khong cch t node n root node t ti mt gi tr no . Vic hn ch chiu su ca tree ny lm gim phc tp ca tree v phn no gip trnh overfitting.

  • nu tng s leaf node vt qu mt ngng no .

  • nu vic phn chia node khng lm gim entropy qu nhiu (information gain nh hn mt ngng no ).

class TreeNode(object):
    def __init__(self, ids=None, children=[], entropy=0, depth=0):
        self.ids = ids  # Danh sch cc ch s d liu trong nt ny
        self.entropy = entropy  #  o entropy ca nt, s c tnh sau
        self.depth = depth  #  su ca nt so vi nt gc
        self.split_attribute = None  # Thuc tnh c chn  chia nh, nu khng phi l nt l
        self.children = children  # Danh sch cc nt con ca nt hin ti
        self.order = None  # Th t cc gi tr ca thuc tnh chia nh trong cc nt con
        self.label = None  # Nhn ca nt nu n l nt l
        
    def set_properties(self, split_attribute, order):
        self.split_attribute = split_attribute  
        self.order = order  

    def set_label(self, label):
        self.label = label

def entropy(freq):
    freq_0 = freq[np.array(freq).nonzero()[0]]  # Loi b tn s bng 0
    prob_0 = freq_0 / float(freq_0.sum())  # Tnh xc sut ca mi tn s
    return -np.sum(prob_0 * np.log(prob_0))  # Tnh entropy

Cui cng l lp DecisionTreeID3 minh ha cch xy dng mt Decision Tree bng thut ton ID3, n s gm cc siu tham s sau trnh overfitting (tng ng nhng g trnh by phn trc):

  • max_depth: gii hn su ca cy.
  • min_samples_split: s mu ti thiu cn c mi node con khi p dng chia thuc tnh node cha.
  • min_gain: gi tr gain ti thiu p dng mt cch phn chia mt node.
  • max_bin: s lng ngng ti a c to ra chia tch mt thuc tnh numeric (s c trnh by k hn di). Lp ny gm cc phng thc quan trng:
  • fit(data, target): Xy dng cy da trn d liu hun luyn l data v nhn tng ng l target. N s ly node gc (root) l node cha ton b d liu v bt u phn chia da vo hm split.
  • split(node): Chia mt Node bng thut ton ID3. N s duyt qua tng thuc tnh (c trng) ca d liu v tnh ton gain cho mi thuc tnh theo cch c trnh by phn trc. Cui cng, n s chn thuc tnh m sau khi s dng chia th c gain cao nht v tr v danh sch cc nt con c chia.

C mt ch thm rng, trong khi chia:

  • Nu thuc tnh l categorical th gi tr thuc tnh ti cc node con chnh l cc gi tr duy nht c th c ca thuc tnh trong b d liu.
  • Nu thuc tnh l numeric th s dng cc ngng phn chia cc d liu vo cc nt con (v d <7, >=7 & <8, etc.). Ngng c chn chnh l trung bnh cng 2 gi tr lin tip trong tp gi tr c sp xp ca thuc tnh ny. Tuy nhin, nu s lng ngng c to ra vt qua max_bin th cc ngng s c chn ngu nhin m bo c nhiu nht max_bin ngng.

class DecisionTreeID3(object):
    def __init__(self, max_depth=8, min_samples_split=2, min_gain=1e-4, max_bin=8):
        self.root = None  # Nt gc ca cy quyt nh
        self.max_depth = max_depth  #  su ti a ca cy
        self.min_samples_split = min_samples_split  # S mu ti thiu  chia nh mt nt
        self.Ntrain = 0  # S lng mu trong tp hun luyn
        self.min_gain = min_gain  # Mc gain ti thiu  chia nh tip tc
        self.max_bin = max_bin # S lng ngng ti a  phn chia d liu lin tc
    
    def fit(self, data, target):
        self.Ntrain = data.count()[0]  # m s lng mu trong tp d liu
        self.data = data  
        self.attributes = list(data)  # Ly danh sch cc thuc tnh
        self.target = target  
        self.labels = target.unique()  # Ly cc nhn mc tiu duy nht
        
        ids = range(self.Ntrain)  # To danh sch cc ch s mu d liu
        self.root = TreeNode(ids=ids, entropy=self._entropy(ids), depth=0)  # To nt gc vi ton b d liu
        queue = [self.root]  
        while queue:
            node = queue.pop()  
            # Nu cha t ti  su gii hn hoc entropy vn ln hn mc thng tin nh nht  chia
            if node.depth < self.max_depth or node.entropy < self.min_gain:
                node.children = self._split(node)  
                if not node.children:  # Nu khng c nt con, t nhn cho nt
                    self._set_label(node)
                queue += node.children  # Thm cc nt con vo hng i
            else:
                self._set_label(node)  # Nu khng th chia nh thm, t nhn cho nt
                
    def _entropy(self, ids):
        if len(ids) == 0:
            return 0
        freq = np.array(self.target.iloc[ids].value_counts())  # m tn s ca cc nhn mc tiu
        return entropy(freq)  # Tnh entropy da trn tn s
    
    def _set_label(self, node):
        target_ids = node.ids
        node.set_label(self.target.iloc[target_ids].mode()[0])  # t nhn cho nt l bng gi tr mode ca cc nhn mc tiu
    
    def _split(self, node):
        ids = node.ids  # Ly danh sch cc ch s d liu trong nt hin ti
        best_gain = 0  
        best_splits = []  # Khi to danh sch cc cch chia nh tt nht
        best_attribute = None 
        order = None  
        sub_data = self.data.iloc[ids, :]  # Ly tp d liu con tng ng vi cc ch s d liu ca nt hin ti

        for i, att in enumerate(self.attributes):  # Duyt qua tng c trng 
            values = sub_data[att].unique()  
            if len(values) == 1:  # Nu c trng ch c mt gi tr duy nht, b qua c trng ny
                continue
            if np.issubdtype(values.dtype, np.number):  # Kim tra nu thuc tnh l lin tc
                # Ly cc ngng chia 
                sorted_values = np.sort(values)
                thresholds = [(sorted_values[j] + sorted_values[j + 1]) / 2 for j in range(len(sorted_values) - 1)]
                # Gii hn ngng chia
                if len(thresholds) > self.max_bin:
                    thresholds = np.random.choice(thresholds, self.max_bin, replace=False)
                    
                for threshold in thresholds:  # Duyt qua cc ngng  kim tra
                    splits = [
                        [self.data.index.get_loc(idx) for idx in sub_data.index[sub_data[att] <= threshold]],  # D liu c gi tr <= ngng
                        [self.data.index.get_loc(idx) for idx in sub_data.index[sub_data[att] > threshold]]  # D liu c gi tr > ngng
                    ]
                    if min(map(len, splits)) < self.min_samples_split:  # Kim tra nu mt trong cc cch chia nh c s lng mu nh hn min_samples_split
                        continue
                    HxS = 0  # Khi to tng entropy c trng s
                    for split in splits:  
                        HxS += len(split) * self._entropy(split) / len(ids)  # Tnh entropy c trng s ca cch chia nh hin ti
                    gain = node.entropy - HxS  # Tnh gain cho cch chia nh hin ti
                    if gain < self.min_gain:  
                        continue
                    if gain > best_gain:  
                        best_gain = gain
                        best_splits = splits
                        best_attribute = att
                        order = [f"<= {threshold}", f"> {threshold}"]
            else:  # X l cc thuc tnh categorical
                splits = []  
                for val in values:  
                    sub_ids = sub_data.index[sub_data[att] == val].tolist()  # Ly danh sch cc ch s d liu c gi tr hin ti ca c trng
                    splits.append([self.data.index.get_loc(idx) for idx in sub_ids])  # Thm danh sch ch s vo danh sch cc cch chia nh
                if min(map(len, splits)) < self.min_samples_split:  
                    continue
                HxS = 0  # Khi to tng entropy c trng s
                for split in splits:  # Duyt qua tng cch chia nh
                    HxS += len(split) * self._entropy(split) / len(ids)  # Tnh entropy c trng s ca cch chia nh hin ti
                gain = node.entropy - HxS  # Tnh gain cho cch chia nh hin ti
                if gain < self.min_gain:  
                    continue
                if gain > best_gain:  
                    best_gain = gain
                    best_splits = splits
                    best_attribute = att
                    order = values
                    
        node.set_properties(best_attribute, order)  # Thit lp thuc tnh chia nh v th t gi tr tt nht cho nt hin ti
        child_nodes = [
            TreeNode(ids=split, entropy=self._entropy(split), depth=node.depth + 1)  
            for split in best_splits
        ]
        return child_nodes  # Tr v danh sch cc nt con


    def predict(self, new_data):
        """
        :param new_data: a new dataframe, each row is a datapoint
        :return: predicted labels for each row
        """
        npoints = new_data.count()[0]
        labels = [None] * npoints
        for n in range(npoints):
            x = new_data.iloc[n, :]  # Ly mt im d liu 
            node = self.root
            while node.children:
                if isinstance(node.order[0], str) and "<=" in node.order[0]:  # Kim tra nu thuc tnh l lin tc
                    threshold = float(node.order[0].split(" ")[1])  # Tch ngng t chui
                    if x[node.split_attribute] <= threshold:
                        node = node.children[0]  # i xung nhnh tri nu gi tr nh hn hoc bng ngng
                    else:
                        node = node.children[1]  # i xung nhnh phi nu gi tr ln hn ngng
                else:
                    node = node.children[node.order.index(x[node.split_attribute])]  # X l thuc tnh categorical
            labels[n] = node.label  
            
        return labels

Phn tch v nh gi kt qu

Trong phn ny, v d liu b mt cn bng (71% mang nhn 0 v 29% mang nhn 1) nn nhm s s dng 2 ch s khc nhau o lng chnh xc ca m hnh:

  • Accuracy: t l s lng d on ng trn tng s d on \begin{align*} accuracy = \frac{TP+TN}{TP+TN+FP+FN} \end{align*}
  • F1 Score: \begin{align*} f1_score = 2\cdot\frac{precision\cdot recall}{precision+recall} \end{align*} Trong precison l thc o mc chnh xc ca cc d on dng tnh ca m hnh: $precision=\frac{TP}{TP+FP}$. V recall l thc o mc y ca cc d on dng tnh ca m hnh $recall=\frac{TP}{TP+FN}$.

Trong cc cng thc trn

  • TP: True Positives (S lng d on ng m thc t cng ng)

  • TN: True Negatives (S lng d on sai m thc t cng sai)

  • FP: False Positives (S lng d on ng m thc t sai)

  • FN: False Negatives (S lng d on sai m thc t ng)

Ngoi ra, nhm cng s xem xt tc ng ca cc siu tham s c a vo v vai tr ca n trong gim thiu overfitting
